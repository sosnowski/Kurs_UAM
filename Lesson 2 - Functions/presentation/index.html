Functions

Functions are the very essence of JavaScript. The language is build around them.

Defining the function


function funcName() {
	//function declaration
}

var func = function () {
	//function expression
}

(function () {
	//IIFE
}())

Return values


var sum = function (arg1, arg2) {
	return arg1 + arg2;
};

sum(5, 4); // 9

arguments


var sumAll = function () {
	var sum = 0;
	for (var i = 0; i < arguments.length; i++) {
	    sum += arguments[i];
	}
	return sum;
};

Closure


var times = function (t) {
	return function (number) {
	    return t * number;
	};
};

var timesTwo = times(2);
timesTwo(5); // 10
timesTwo(4); //8

var timesThree = times(3);
timesThree(5); // 15
timesThree(4); // 12

Function context

Is this the value of this?


var obj {
	method: function () {
	    console.log(this);
	}
}
obj.method(); // obj

var func = function () {
	console.log(this);
};

func(); // window (global object);

Context is dynamic, depends on how the function is called


obj.method(); // obj
var func = obj.method;

func(); // window

Every function has its own context


var obj = {
	method: function () {
	    console.log(this);

	    (function () {
	        console.log(this);
	    }());
	}
}
obj.method(); // obj; window;

Context can be changed


var obj = {}, func = function () {
	console.log(this);
};

func(); // window
func.apply(obj); //obj
func.call(obj); //obj

func(); //window

var func2 = func.bind(obj);
func2(); // obj
func2(); // obj

Common problems


var form = {
	checkSmth: function () { //... },
	addSubmitAction: function () {
	    // ...
	    formEl.addEventListener('submit', function () {
	        this.validateData(); // !
	    });
	},
	validateData: function () {
	    //fancy validation here
	    this.checkSmth();
	}
}

Solutions


var form = {
   checkSmth: function () { //... },
   addSubmitAction: function () {
      formEl.addEventListener('submit', this.validateData.bind(this));
    },
    validateData: function () {
       this.checkSmth();
    }
}

var form = {
   checkSmth: function () { //... },
   addSubmitAction: function () {
      var me = this;
      formEl.addEventListener('submit', function () {
          me.validateData();
      });
   },
   validateData: function () {
      me.checkSmth();
   }
}
